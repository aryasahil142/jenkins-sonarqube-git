pipeline {
    agent any
    stages {
        stage ("Pull") {
            steps {
                git 'https://github.com/aryasahil142/sonar-jenkins.git'
            }
        }
        stage ("Build") {
            steps {
                sh '/opt/maven/bin/mvn clean package'
            }
        }
        stage ("Test") {
            steps {
              withSonarQubeEnv(credentialsId: 'my-sonar-token', installationName: 'my-sonar') {
                sh'''/opt/maven/bin/mvn verify org.sonarsource.scanner.maven:sonar-maven-plugin:sonar \
                      -Dsonar.projectKey=java-student '''
}
            }
        }
       stage ("Quality Gate") {
            steps {
               timeout(time: 5, unit: 'MINUTES') {
    waitForQualityGate abortPipeline: true, credentialsId: 'my-sonar-token'
}
}
}
        stage ("Deploy") {
            steps {
              deploy adapters: [tomcat9(alternativeDeploymentContext: '', credentialsId: 'tom-cruise', path: '', url: 'http://13.234.34.2:8080/')], contextPath: 'student', war: '**.war'
            }
        }
    }
}
